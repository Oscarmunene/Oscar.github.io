// Get employee details from user const employeeName = prompt('Enter employee name:'); const hoursWorked = parseFloat(prompt('Enter hours worked:')); const ratePerHour = parseFloat(prompt('Enter rate per hour:')); // Calculate basic pay const basicPay = hoursWorked * ratePerHour; // Calculate tax based on basic pay let tax; if (basicPay > 50000) { tax = basicPay * 0.2; } else if (basicPay >= 20000) { tax = basicPay * 0.1; } else { tax = 0; } // Calculate net pay const netPay = basicPay - tax; // Display results console.log(`Employee Name: ${employeeName}`); console.log(`Basic Pay: ${basicPay.toFixed(2)}`); console.log(`Tax: ${tax.toFixed(2)}`); console.log(`Net Pay: ${netPay.toFixed(2)}`);
